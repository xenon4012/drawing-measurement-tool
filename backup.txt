<div id="toolbar">
  Drag & drop an image. Draw to measure in cm. Right-click to clear.
</div>

<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

// ====== SETTINGS ======
// Change page size here if needed
const PAGE_WIDTH_CM = 16.51;   // 6.5 inches
const PAGE_HEIGHT_CM = 22.86;  // 9 inches
// ======================

let drawing = false;
let startX = 0;
let startY = 0;
let lines = [];
let image = null;

// --- Drag & Drop ---
canvas.addEventListener("dragover", (e) => e.preventDefault());

canvas.addEventListener("drop", (e) => {
  e.preventDefault();

  const file = e.dataTransfer.files[0];
  if (!file || !file.type.startsWith("image/")) return;

  const img = new Image();
  const reader = new FileReader();

  reader.onload = function(event) {
    img.onload = function() {
      image = img;

      canvas.width = img.width;
      canvas.height = img.height;

      lines = [];
      redraw();
    };
    img.src = event.target.result;
  };

  reader.readAsDataURL(file);
});

// --- Drawing ---
canvas.addEventListener("mousedown", (e) => {
  if (e.button === 2) return; // ignore right-click

  drawing = true;
  const rect = canvas.getBoundingClientRect();
  startX = e.clientX - rect.left;
  startY = e.clientY - rect.top;
});

canvas.addEventListener("mousemove", (e) => {
  if (!drawing) return;

  const rect = canvas.getBoundingClientRect();
  let currentX = e.clientX - rect.left;
  let currentY = e.clientY - rect.top;

  redraw();

  const dx = Math.abs(currentX - startX);
  const dy = Math.abs(currentY - startY);

  if (dx > dy) {
    currentY = startY;
  } else {
    currentX = startX;
  }

  drawLine(startX, startY, currentX, currentY, true);
});

canvas.addEventListener("mouseup", (e) => {
  if (!drawing) return;
  drawing = false;

  const rect = canvas.getBoundingClientRect();
  let endX = e.clientX - rect.left;
  let endY = e.clientY - rect.top;

  const dx = Math.abs(endX - startX);
  const dy = Math.abs(endY - startY);

  if (dx > dy) {
    endY = startY;
  } else {
    endX = startX;
  }

  lines.push({ x1: startX, y1: startY, x2: endX, y2: endY });

  redraw();
});

// --- Right-click to clear ---
canvas.addEventListener("contextmenu", (e) => {
  e.preventDefault();
  lines = [];
  redraw();
});

// --- Drawing helpers ---
function drawLine(x1, y1, x2, y2, preview = false) {
  ctx.beginPath();
  ctx.moveTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.strokeStyle = preview ? "red" : "black";
  ctx.lineWidth = preview ? 3 : 2;
  ctx.stroke();

  // ❗ Skip label while drawing (preview mode)
  if (preview) return;

  // Pixel length
  const pixelLength = Math.abs(x2 - x1) + Math.abs(y2 - y1);

  // Convert to cm
  const cmPerPixelX = PAGE_WIDTH_CM / canvas.width;
  const cmPerPixelY = PAGE_HEIGHT_CM / canvas.height;

  let lengthCM = (x1 === x2)
    ? pixelLength * cmPerPixelY
    : pixelLength * cmPerPixelX;

  const text = `${lengthCM.toFixed(2)} cm`;
  ctx.font = "16px Arial";

  const textWidth = ctx.measureText(text).width;
  const textHeight = 16;

  let textX = (x1 + x2) / 2;
  let textY = (y1 + y2) / 2 - 5;

  // Clamp inside canvas
  const padding = 6;

  if (textX - textWidth / 2 < padding) {
    textX = textWidth / 2 + padding;
  }
  if (textX + textWidth / 2 > canvas.width - padding) {
    textX = canvas.width - textWidth / 2 - padding;
  }
  if (textY - textHeight < padding) {
    textY = textHeight + padding;
  }
  if (textY > canvas.height - padding) {
    textY = canvas.height - padding;
  }

  // Background box
  ctx.fillStyle = "rgba(255, 255, 255, 0.85)";
  ctx.fillRect(
    textX - textWidth / 2 - 4,
    textY - textHeight + 4,
    textWidth + 8,
    textHeight + 4
  );

  // Text
  ctx.fillStyle = "black";
  ctx.fillText(text, textX - textWidth / 2, textY);
}

function redraw() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  if (image) {
    ctx.drawImage(image, 0, 0);
  }

  for (let line of lines) {
    drawLine(line.x1, line.y1, line.x2, line.y2);
  }
}
</script>
